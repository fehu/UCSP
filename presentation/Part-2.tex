        \resizeinput{\rootdir/img/NQueens/Restrictions-content.tikz}

      \begin{column}{.6\textwidth}
        \centering
        \resizebox{\textwidth}{!}{%
        $\xi_2(\left<x_1,y_1\right>, \left<x_2,y_2\right>) =
            \begin{cases}
              0 & \mbox{if } \begin{cases}
                                &      x_1 = x_2  \lor y_1 = y_2 \\
                                \lor~& x_1 = y_2 \land x_2 = y_1 \\
                                \lor~& x_1-y_1 = x_2-y_2\\
                                \lor~& x_1+y_1 = x_2+y_2
                             \end{cases} \\
             1 & \text{otherwise}
            \end{cases}
        $}%
      \end{column}
    \end{columns}
  \end{examples}
\end{frame}
