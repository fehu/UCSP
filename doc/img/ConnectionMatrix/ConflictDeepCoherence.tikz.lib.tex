\newcommand*{\showExtCoh}[5][]{
  \def\m{#2}
  \def\a{#3}
  \def\d{#4}
  \def\pss{#5}

  \gdef\cohs{}
  \gdef\first{1}
  \foreach \ps in \pss
    { \ifthenelse{\first=1}
          {\numgdef\first{0}}
          {\xappto\cohs{\noexpand\\[-4pt]}}
      \foreach \x in \ps {\xappto\cohs{, coh[\x]}}
     }
  \node[below=1em of \m, text width=9cm, align=center, #1]
      {External coherence of depth \d\ by $\a$:
       \begin{gather*}
         \Gamma\big( coh[\a] \cohs \big)
       \end{gather*}
       };
}
